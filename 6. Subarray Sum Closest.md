# Description

Given an integer array, find a subarray with sum closest to zero. Return the indexes of the first number and last number.

class Solution {
public:
    /*
     * @param nums: A list of integers
     * @return: A list of integers includes the index of the first number and the index of the last number
     */
    
    vector<int> subarraySumClosest(vector<int> &nums) {
        // write your code here
        vector<int> sum(nums.size(), 0);
        vector<int> ans;
        int diff = INT_MAX;
        for (int i = 0; i < nums.size(); ++i) {
            if (i == 0)
                sum[i] = nums[i];
            else
                sum[i] = sum[i - 1] + nums[i];
            if (abs(sum[i]) <= diff) {
                diff = abs(sum[i]);
                ans = {0, i};
            }
        }
        for (int i = 0; i < sum.size() - 1; ++i) {
            for (int j = i + 1; j < sum.size(); ++j) {
                if (abs(sum[j] - sum[i]) < diff) {
                    diff = abs(sum[j] - sum[i]);
                    ans = {i + 1, j};
                }
            }
        }
        return ans;
    }
};

### method 2

- O(nlogn)
- struct (overwrite operator <, constructor)

class Solution {
public:
    /*
     * @param nums: A list of integers
     * @return: A list of integers includes the index of the first number and the index of the last number
     */
    struct T {
        //T() {};
        T(int v, int p): val(v), pos(p) {};
        int val, pos;
        bool operator < (const T& x)const { 
            return val < x.val || val == x.val && pos < x.pos; 
        }
    };
    vector<int> subarraySumClosest(vector<int> &nums) {
        vector<T> s;
        // s.reserve(nums.size());
        vector<int> ans = {0,0};
        int sum = 0;
        s.push_back(T(0,-1));
        for (int i = 0; i < nums.size(); ++i) {
            sum += nums[i];
            s.push_back(T(sum, i));
        }
        sort(s.begin(), s.end());
        int ss = INT_MAX;
        for (int i = 0; i < nums.size() - 1; ++i) {
            if (s[i+1].val - s[i].val < ss) {
                ss = s[i+1].val - s[i].val;
                ans = {min(s[i].pos, s[i+1].pos) + 1, max(s[i].pos, s[i+1].pos)};
            }
        }
        return ans;
    }
};
